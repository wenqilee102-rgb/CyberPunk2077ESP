{
	public    var meshesAppearence : CName;

	function Execute(owner : whandle:gameObject); // Execute;GameObject
	function ResolveMeshesAppearence(appearanceName : CName, owner : whandle:gameObject); // ResolveMeshesAppearence;CNameGameObject
	function Restore(owner : whandle:gameObject); // Restore;GameObject

class abstract DeviceOperationBase extends IScriptable
{
	public    var operationName : CName;
	public    var executeOnce : Bool;
	protected var isEnabled : Bool;
	public    var toggleOperations : array:SToggleDeviceOperationData;
	public    var disableDevice : Bool;

	function Execute(owner : whandle:gameObject); // Execute;GameObject
	function IsEnabled() : Bool; // IsEnabled;
	function ResolveDisable(disable : Bool, owner : whandle:gameObject); // ResolveDisable;BoolGameObject
	function Restore(owner : whandle:gameObject); // Restore;GameObject
	function SetIsEnabled(enabled : Bool); // SetIsEnabled;Bool
}

	import static function DetectScriptableCycles(); // DetectScriptableCycles

	import function IsExactlyA(className : CName) : Bool; // IsExactlyA
